@app.route('/leaves_overview') @login_required def leaves_overview(): if current_user.role not in ('faculty', 'hod'): flash("Unauthorized access", "danger") return redirect(url_for('dashboard')) month = request.args.get('month') date_str = request.args.get('date') # Start with a query joining the requester (student/user) query = Leave.query.join(Leave.requester) # Faculty: can only see their class students' leaves if current_user.role == 'faculty': class_ids = [c.id for c in current_user.assigned_classes] query = query.join(ClassGroup).filter(ClassGroup.id.in_(class_ids)) # HOD: can see all students in their department elif current_user.role == 'hod': query = query.join(ClassGroup).filter( ClassGroup.department_id == current_user.department_id ) # Month filter if month: try: year, month_num = map(int, month.split("-")) query = query.filter( db.extract('year', Leave.start_date) == year, db.extract('month', Leave.start_date) == month_num ) except ValueError: flash("Invalid month format", "danger") # Specific date filter if date_str: try: date = datetime.strptime(date_str, "%Y-%m-%d").date() query = query.filter( Leave.start_date <= date, Leave.end_date >= date ) except ValueError: flash("Invalid date format", "danger") # Get results ordered by newest first leaves = query.order_by(Leave.start_date.desc()).all() return render_template("leaves_overview.html", leaves=leaves)
